User Details schema:

CREATE TABLE `biometric`.`userdetails` (
  `pk` INT NOT NULL AUTO_INCREMENT,
  `name` VARCHAR(45) NOT NULL,
  `phoneNumber` BIGINT(10) NOT NULL,
  `emailId` VARCHAR(45) NULL,
  `address` VARCHAR(45) NULL,
  `age` INT(100) NULL,
  `fpIso` BLOB NOT NULL,
  `fbBmpImage` LONGBLOB NOT NULL,
  UNIQUE INDEX `pk_UNIQUE` (`pk` ASC),
  PRIMARY KEY (`pk`));
  
Card Details Schema:
  CREATE TABLE `biometric`.`carddetails` (
  `pk` INT NOT NULL AUTO_INCREMENT,
  `nameOnCard` VARCHAR(45) NOT NULL,
  `cardNumber` VARCHAR(16) NOT NULL,
  `cvv` INT(3) NOT NULL,
  `expiryDate` VARCHAR(6) NULL,
  `fk` INT NOT NULL,
  PRIMARY KEY (`pk`),
  UNIQUE INDEX `pk_UNIQUE` (`pk` ASC),
  INDEX `userPk_idx` (`fk` ASC),
  CONSTRAINT `userPk`
    FOREIGN KEY (`fk`)
    REFERENCES `biometric`.`userdetails` (`pk`)
    ON DELETE CASCADE
    ON UPDATE CASCADE);
